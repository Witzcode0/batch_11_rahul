#include<stdio.h>
void main(){
    int num = 5;

    for(int row = 1; row<=num; row+=1){
        for(int col = 1; col<=num; col+=1){
            printf("* ");
        }
        printf("\n");
    }
}

#include<stdio.h>
void main(){
    int num = 5;

    for(int row = 1; row<=num; row+=1){
        for(int col = 1; col<=num-row; col+=1){
            printf("  ");
        }
        for(int col = 1; col<=row; col+=1){
            printf("* ");
        }
        printf("\n");
    }
}

#include<stdio.h>
void main(){
    int num = 5;

    for(int row = 1; row<=num; row+=1){
        for(int col = 1; col<=row-1; col+=1){
            printf("  ");
        }
        for(int col = 1; col<=num-row+1; col+=1){
            printf("%c ", col + 64);
        }

        printf("\n");
    }
}

#include<stdio.h>
void main(){
    int num = 5;
    int g = 1;

    for(int row = 1; row<=num; row+=1){
        for(int col = 1; col<=row; col+=1){
            printf("%d ", g);
            g++;
        }
        printf("\n");
    }
}
